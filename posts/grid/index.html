/// @brief ///////////
int tr[N * 3 + 10];

int lowbit(int x)
{
    return x & (-x);
}

void update(int i, int x) // a[i]+=x;
{
    // if(i==0)
    // {
    //     tr[i]+=x;
    //     return;
    // }
    for (int j = i; j <= 3 * m + 5; j += lowbit(j))
    {
        tr[j] += x;
    }
}

int query(int r) // a[0]+..+a[r]
{
    if (r < 0)
        return 0;
    r = min(r, 3 * m + 5);
    int sum = 0;
    for (int i = r; i >= 1; i -= lowbit(i))
    {
        sum += tr[i];
    }
    return sum;
}
